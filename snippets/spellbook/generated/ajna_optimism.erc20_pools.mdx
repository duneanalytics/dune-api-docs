</ParamField>
<ParamField path="quotes">
Quote token address
</ParamField>
<ParamField path="pool_address">
Pool address
</ParamField>
<ParamField path="starting_interest_rate">
Initial interest rate set by deployer
</ParamField>
<ParamField path="tx_hash">
TX hash of deployment
</ParamField>
<ParamField path="block_time">
Time of deployment
</ParamField>
<ParamField path="block_date">
Date of deployment
</ParamField>
<ParamField path="block_number">
Block number that pool was deployed
</ParamField>
