## eas.attestation_details 
EAS attestation details

<ParamField path="blockchain">
Blockchain where EAS is deployed
</ParamField>
<ParamField path="project">
Project name
</ParamField>
<ParamField path="version">
Version of the EAS contract
</ParamField>
<ParamField path="schema_uid">
Schema unique identifier
</ParamField>
<ParamField path="attestation_uid">
Attestation unique identifier
</ParamField>
<ParamField path="ordinality">
Incremental unique number to order fields within each Schema
</ParamField>
<ParamField path="data_type">
Schema field data type. Schemas follow Solidity ABI for acceptable data types.
</ParamField>
<ParamField path="field_name">
Name of Schema field
</ParamField>
<ParamField path="decoded_data">
Decoded data chunk (if successful otherwise blank) related to data_type/field_name pair
</ParamField>
<ParamField path="is_revoked">
Is this Attestation revoked?
</ParamField>
<ParamField path="block_number">
Block number
</ParamField>
<ParamField path="block_time">
UTC event block time
</ParamField>
<ParamField path="tx_hash">
Unique transaction hash value
</ParamField>
<ParamField path="evt_index">
Index of the corresponding transaction event
</ParamField>
