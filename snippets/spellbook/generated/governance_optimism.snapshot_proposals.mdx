## governance_optimism.snapshot_proposals 
Analysis of optimism governance proposals available on snapshot platform 


<ParamField path="proposal_id">
The unique identifier of the proposal
</ParamField>
<ParamField path="proposal_link">
The link of the proposal on respective platform
</ParamField>
<ParamField path="proposal_type">
The type of proposal
</ParamField>
<ParamField path="proposal_description">
A brief description of the proposal
</ParamField>
<ParamField path="proposal_created_at">
Block time when a proposal was created onchain
</ParamField>
<ParamField path="start_block">
The block number when the voting started
</ParamField>
<ParamField path="start_timestamp">
The date and time when the proposal voting started
</ParamField>
<ParamField path="end_block">
The block number when the voting ended
</ParamField>
<ParamField path="end_timestamp">
The date and time when the proposal voting ended
</ParamField>
<ParamField path="platform">
The decentralized governance platform on which the proposal was created (e.g., Agora, Snapshot)
</ParamField>
<ParamField path="highest_weightage_vote">
The vote with the highest weight
</ParamField>
<ParamField path="highest_weightage_voter">
The voter whose vote is the highest weight
</ParamField>
<ParamField path="highest_weightage_voter_percentage">
The percentage of the total voting weight that the highest weight voter has
</ParamField>
<ParamField path="total_for_votingWeightage">
The total voting weight of all the votes in favor of the proposal
</ParamField>
<ParamField path="total_abstain_votingWeightage">
The total voting weight of all the votes that abstained on the proposal
</ParamField>
<ParamField path="total_against_votingWeightage">
The total voting weight of all the votes against the proposal
</ParamField>
<ParamField path="unique_for_votes">
The unique number of votes in favor of the proposal
</ParamField>
<ParamField path="unique_abstain_votes">
The unique number of votes that abstained on the proposal
</ParamField>
<ParamField path="unique_against_votes">
The unique number of votes against the proposal
</ParamField>
<ParamField path="unique_votes_count">
The unique number of votes cast on the proposal
</ParamField>
<ParamField path="total_votes_casted">
The total number of votes cast on the proposal, excluding abstaining votes
</ParamField>
<ParamField path="unique_voters">
The total number of distinct voter addresses that cast votes on the proposal
</ParamField>
<ParamField path="proposal_status">
The current status of the proposal
</ParamField>
