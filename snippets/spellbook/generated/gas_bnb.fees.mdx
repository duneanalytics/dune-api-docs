## gas_bnb.fees 
Gas Fees on BNB chain


<ParamField path="blockchain">
Blockchain name
</ParamField>
<ParamField path="block_number">
Block number
</ParamField>
<ParamField path="block_time">
Timestamp for block event time in UTC
</ParamField>
<ParamField path="tx_hash">
Primary key of the transaction
</ParamField>
<ParamField path="tx_sender">
Initiated the transaction
</ParamField>
<ParamField path="tx_receiver">
Received the transaction
</ParamField>
<ParamField path="native_token_symbol">
Symbol of native token (BNB) for the BNB chain
</ParamField>
<ParamField path="tx_amount_native">
Amount of native tokens (BNB) transferred from sender to recipient
</ParamField>
<ParamField path="tx_amount_usd">
Amount of $USD transferred from sender to recipient
</ParamField>
<ParamField path="tx_fee_native">
Total amount of native tokens (BNB) paid in gas fees -- Gas Price * Gas Used
</ParamField>
<ParamField path="tx_fee_usd">
Total amount of $USD paid in gas fees
</ParamField>
<ParamField path="burned_native">
Total amount of native tokens (BNB) burned for this transaction, Post BEP-95: 10% of transaction fee
</ParamField>
<ParamField path="burned_usd">
Total amount of $USD burned for this transaction, Post BEP-95: 10% of transaction fee
</ParamField>
<ParamField path="validator">
Address of blockchain validator for this transaction within the block
</ParamField>
<ParamField path="gas_price_gwei">
Gas price (per gas unit) denoted in gwei
</ParamField>
<ParamField path="gas_price_usd">
Gas price (per gas unit) denoted in $USD
</ParamField>
<ParamField path="gas_used">
Amount of gas units consumed by the transaction
</ParamField>
<ParamField path="gas_limit">
Maximum amount of gas units that can be consumed by the transaction
</ParamField>
<ParamField path="gas_usage_percent">
Percentage of Gas Used relative to the gas limit
</ParamField>
<ParamField path="transaction_type">
Transaction type
</ParamField>
