</ParamField>
<ParamField path="symbol">
Token symbol
</ParamField>
<ParamField path="evt_block_hour">

</ParamField>
<ParamField path="deposit_apy">

</ParamField>
<ParamField path="stable_borrow_apy">
Hourly stable borrow interest rate value of the reserve
</ParamField>
<ParamField path="variable_borrow_apy">

</ParamField>
