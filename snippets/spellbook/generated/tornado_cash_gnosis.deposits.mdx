## tornado_cash_gnosis.deposits 
Tornado Cash Deposits on Gnosis

<ParamField path="block_time">
UTC event block time
</ParamField>
<ParamField path="currency_contract">
Currency contract address (on Ethereum mainnet) of currency deposited/withdrawn from Tornado Cash, with ETH contract address swapped for WETH
</ParamField>
<ParamField path="currency_symbol">
Symbol of currency deposited/withdrawn from Tornado Cash
</ParamField>
<ParamField path="blockchain">
Blockchain
</ParamField>
<ParamField path="tornado_version">
Version of Tornado Cash used for this transaction
</ParamField>
<ParamField path="depositor">
Address depositing into Tornado Cash
</ParamField>
<ParamField path="contract_address">
Contract Address of used Tornado Cash Pool
</ParamField>
<ParamField path="leaf_index">
Leaf index
</ParamField>
<ParamField path="evt_index">
Event index
</ParamField>
