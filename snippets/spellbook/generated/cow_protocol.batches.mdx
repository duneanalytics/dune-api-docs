## cow_protocol.batches 
CoW Protocol enriched batches table cross-chain


<ParamField path="blockchain">
Blockchain which the project is deployed
</ParamField>
<ParamField path="project">
Project name
</ParamField>
<ParamField path="version">
Version of the contract built and deployed by the project
</ParamField>
<ParamField path="block_date">
UTC event block date of each trade
</ParamField>
<ParamField path="block_time">
UTC event block time of each trade
</ParamField>
<ParamField path="num_trades">
Number of trades settled within the batch
</ParamField>
<ParamField path="dex_swaps">
Number of AMM interactions used to settle the batch (estimated)
</ParamField>
<ParamField path="batch_value">
Total USD value of trades settled within the batch
</ParamField>
<ParamField path="solver_address">
Ethereum address of solver settled batch (equivalent to tx.from)
</ParamField>
<ParamField path="tx_hash">
Unique transaction hash value tied to each transaction on the DEX
</ParamField>
<ParamField path="gas_price">
Gas price (in WEI) of transaction
</ParamField>
<ParamField path="gas_used">
Total/cumulative gas consumed (in WEI) to settle the batch
</ParamField>
<ParamField path="tx_cost_usd">
USD cost of batch execution (gas_price * gas_used) / 1e18 * ETH price
</ParamField>
<ParamField path="fee_value">
Total USD valuation of fees collected from all trades within the batch
</ParamField>
<ParamField path="call_data_size">
Size (in Kilobytes) of call data passed into the settle method.
</ParamField>
<ParamField path="unwraps">
Number of WETH unwraps performed within the settlement
</ParamField>
<ParamField path="token_approvals">
Number of ERC20 token approvals made within the settlement
</ParamField>
