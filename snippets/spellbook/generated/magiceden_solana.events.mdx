</ParamField>
<ParamField path="royalty_fee_amount_raw">

</ParamField>
<ParamField path="royalty_fee_amount">
Royalty fee amount in original token currency (properly formatted in decimals)
</ParamField>
<ParamField path="royalty_fee_amount_usd">
Royalty fee amount in USD
</ParamField>
<ParamField path="royalty_fee_percentage">
Royalty fee in % of the amount paid for a given trade
</ParamField>
<ParamField path="royalty_fee_currency_symbol">
Symbol of the token in which fees are paid out
</ParamField>
<ParamField path="unique_trade_id">
Unique trade id, combination of signatures[0] and id
